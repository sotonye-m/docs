---
title: "Submit Individual KYC"
description: "Submit identity verification data for an individual customer"
---

Submit KYC (Know Your Customer) data for an individual customer. The customer must already exist in your account. After submission, the status will be `submitted` and move to `approved` after review.

```
POST https://kyc.yativo.com/api/individual-kyc/submit
```

<Note>
  This endpoint is on the KYC platform (`https://kyc.yativo.com`), not the main API base URL. Requires `Idempotency-Key` header.
</Note>

## Identity fields

<ParamField body="customer_id" type="string" required>
  The customer UUID from `POST /customer`.
</ParamField>

<ParamField body="type" type="string" required>
  Must be `"individual"`.
</ParamField>

<ParamField body="signed_agreement_id" type="string">
  Optional agreement UUID. If omitted, it is auto-generated by Yativo.
</ParamField>

<ParamField body="first_name" type="string" required>
  Customer's first name.
</ParamField>

<ParamField body="last_name" type="string" required>
  Customer's last name.
</ParamField>

<ParamField body="middle_name" type="string">
  Customer's middle name (optional).
</ParamField>

<ParamField body="email" type="string" required>
  Customer's email address.
</ParamField>

<ParamField body="phone" type="string" required>
  Phone number without country code.
</ParamField>

<ParamField body="calling_code" type="string" required>
  Country calling code, e.g. `"+1"`, `"+44"`, `"+55"`.
</ParamField>

<ParamField body="gender" type="string" required>
  One of: `"male"`, `"female"`, `"other"`.
</ParamField>

<ParamField body="birth_date" type="string" required>
  Date of birth in `YYYY-MM-DD` format.
</ParamField>

<ParamField body="nationality" type="string" required>
  ISO 3166-1 alpha-2 country code (e.g. `"US"`, `"BR"`, `"GB"`).
</ParamField>

<ParamField body="taxId" type="string" required>
  Tax identification number for the customer.
</ParamField>

<ParamField body="selfie_image" type="string" required>
  URL or base64-encoded selfie image of the customer.
</ParamField>

    <ParamField body="city" type="string" required>City.</ParamField>
    <ParamField body="state" type="string" required>State or province.</ParamField>
    <ParamField body="postal_code" type="string" required>Postal or ZIP code.</ParamField>
    <ParamField body="country" type="string" required>ISO 3166-1 alpha-2 country code.</ParamField>
    <ParamField body="proof_of_address_file" type="string" required>URL or base64-encoded proof of address document.</ParamField>
  </Expandable>
</ParamField>

## Identifying information

<ParamField body="identifying_information" type="array" required>
  Minimum 1 government-issued ID document required.

  <Expandable title="identifying_information item">
    <ParamField body="type" type="string" required>`"passport"`, `"national_id"`, or `"drivers_license"`.</ParamField>
    <ParamField body="issuing_country" type="string" required>ISO 3166-1 alpha-2 issuing country code.</ParamField>
    <ParamField body="number" type="string" required>Document number.</ParamField>
    <ParamField body="date_issued" type="string" required>Issue date in `YYYY-MM-DD` format.</ParamField>
    <ParamField body="expiration_date" type="string" required>Expiration date in `YYYY-MM-DD` format.</ParamField>
    <ParamField body="image_front_file" type="string" required>URL or base64-encoded front image.</ParamField>
    <ParamField body="image_back_file" type="string">URL or base64-encoded back image (required for national_id, drivers_license).</ParamField>
  </Expandable>
</ParamField>

## Financial information

<ParamField body="employment_status" type="string" required>
  One of: `"Employed"`, `"self_employed"`, `"unemployed"`, `"retired"`, `"student"`.
</ParamField>

<ParamField body="most_recent_occupation_code" type="string" required>
  Occupation code from `GET /auth/occupation-codes`.
</ParamField>

<ParamField body="expected_monthly_payments_usd" type="string" required>
  `"LessThan5K"`, `"5KTo50K"`, or `"MoreThan50K"`.
</ParamField>

<ParamField body="source_of_funds" type="string" required>
  Source of funds: `"Salary"`, `"Business"`, `"Investment"`, `"Savings"`, etc.
</ParamField>

<ParamField body="account_purpose" type="string" required>
  Purpose for the account, e.g. `"ReceiveSalary"`.
</ParamField>

<ParamField body="account_purpose_other" type="string">
  Required when `account_purpose` is `"other"`.
</ParamField>

<ParamField body="acting_as_intermediary" type="boolean" required>
  Whether the customer acts as an intermediary for third parties.
</ParamField>

## Virtual account preferences

<ParamField body="usd_virtual_account" type="boolean">
  Request a USD virtual account.
</ParamField>

<ParamField body="eur_virtual_account" type="boolean">
  Request a EUR virtual account.
</ParamField>

<ParamField body="eurde_virtual_account" type="boolean">
  Request a EUR/DE virtual account. If `true`, `usd_virtual_account` and `eur_virtual_account` must be `false`.
</ParamField>

<ParamField body="gbp_virtual_account" type="boolean">
  Request a GBP virtual account.
</ParamField>

  </Expandable>
</ParamField>

<ParamField body="sumsub_token" type="string">
  Optional Sumsub token for submissions that include liveness checks.
</ParamField>
